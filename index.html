import { useState, useCallback } from 'react' let _id = 0 export function useToast() { const [toasts, setToasts] = useState([]) const toast = useCallback((msg, type = '') => { const id = ++_id setToasts(prev => [...prev, { id, msg, type }]) setTimeout(() => setToasts(prev => prev.filter(t => t.id !== id)), 2800) }, []) return { toasts, toast } }
